About

This page documents the process of cleaning the 2010-2025 ARRL Field Day data.

The datasets have different field names across the years, so the field names must be normalized in order to combine the datasets for visualization.

Tidy names

Code 
clean_fd_names_save <- function(fn, clean_file_dir = "data/cleaned") {
  assertthat::is.string(fn)
  assertthat::is.readable(fn)
  
  df <- readr::read_csv(fn, show_col_types = FALSE)
  assertthat::assert_that(tibble::is_tibble(df))
  
  # Clean names
  df_clean <- janitor::clean_names(df)

  fn_no_ext <- basename(tools::file_path_sans_ext(fn))
  fn <- file.path(clean_file_dir, paste0(fn_no_ext, "-clean-names.csv"))
  readr::write_csv(df_clean, fn)
  fn
}

show_names <- function(fn) {
  assertthat::is.string(fn)
  assertthat::is.readable(fn)
  
  df <- readr::read_csv(fn, show_col_types = FALSE)
  assertthat::assert_that(tibble::is_tibble(df))
  file_year <- stringr::str_extract(fn, pattern = "[0-9]{4}")
  sprintf("%s\n", paste0("--- ", file_year, " ---"))
  names(df)
}

# Clean

## About

This page documents the process of cleaning the 2010-2025 ARRL Field Day data.

The datasets have different field names across the years, so the field names must be normalized in order to combine the datasets for visualization.

## Tidy names

```{r}
clean_fd_names_save <- function(fn, clean_file_dir = "data/cleaned") {
  assertthat::is.string(fn)
  assertthat::is.readable(fn)
  
  df <- readr::read_csv(fn, show_col_types = FALSE)
  assertthat::assert_that(tibble::is_tibble(df))
  
  # Clean names
  df_clean <- janitor::clean_names(df)

  fn_no_ext <- basename(tools::file_path_sans_ext(fn))
  fn <- file.path(clean_file_dir, paste0(fn_no_ext, "-clean-names.csv"))
  readr::write_csv(df_clean, fn)
  fn
}
```

```{r}
show_names <- function(fn) {
  assertthat::is.string(fn)
  assertthat::is.readable(fn)
  
  df <- readr::read_csv(fn, show_col_types = FALSE)
  assertthat::assert_that(tibble::is_tibble(df))
  file_year <- stringr::str_extract(fn, pattern = "[0-9]{4}")
  sprintf("%s\n", paste0("--- ", file_year, " ---"))
  names(df)
}
```

## Merge cleaned

```{r}
import_add_year <- function(fn) {
  assertthat::is.string(fn)
  assertthat::is.readable(fn)
  
  df <- readr::read_csv(fn, show_col_types = FALSE)
  assertthat::assert_that(tibble::is_tibble(df))
  df$year <- stringr::str_extract(fn, "[0-9]{4}")
  df
}

fl <- list.files("data/cleaned", pattern = "normed", full.names = TRUE)

fd_all_df <- purrr::map(fl, import_add_year) |>
  purrr::list_rbind()
```
